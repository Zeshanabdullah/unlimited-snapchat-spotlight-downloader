### PHP Example

This example uses cURL to download a Spotlight video.

php
<?php

function download_snapchat_video($video_url, $save_path) {
    $ch = curl_init($video_url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);

    $data = curl_exec($ch);
    if(curl_errno($ch)) {
        echo 'Error: ' . curl_error($ch);
    } else {
        file_put_contents($save_path, $data);
        echo "Video downloaded successfully!";
    }
    
    curl_close($ch);
}

$video_url = "https://hdstockimages.com/snapchat-spotlight-downloader/video-url";

async function downloadSnapchatVideo(videoUrl, savePath) {
    try {
        const response = await fetch(videoUrl);
        if (!response.ok) throw new Error('Network response was not ok');

        const blob = await response.blob();
        const objectUrl = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = objectUrl;
        a.download = savePath;
        a.click();
        console.log("Video downloaded successfully!");
    } catch (error) {
        console.error('An error occurred:', error);
    }
}
